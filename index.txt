        // --- Link đã được cập nhật ---
        const shopeeLink = 'https://s.shopee.vn/8Uz9oBxXLE';
        const googleSheetLink = 'https://docs.google.com/spreadsheets/d/1YCib2JSf68lwcfq7eDpf9FqJqCJX219q5-t61KL-5X4/edit?gid=1996725396#gid=1996725396';

        // Thời gian chờ trước khi chuyển hướng (tính bằng mili giây, 5000ms = 5 giây)
        // Bạn có thể thay đổi số 5000 thành 7000 (7 giây) nếu muốn
        const delayInMilliseconds = 5000;

        // 1. Cố gắng mở link Shopee ngay lập tức
        window.location.href = shopeeLink;

        // 2. Đặt lịch chuyển hướng đến Google Sheet sau khoảng thời gian chờ
        setTimeout(function() {
            window.location.href = googleSheetLink;
        }, delayInMilliseconds);
